막대 그래프

개요

사용법

막대그래프를 사용하려면 tsart-bar-standard.js 파일을 참조해야 한다.

<!-- Tsart 라이브러리 참조 -->
<script src="...path.../tsart.js"></script>
<script src="...path.../tsart-bar-standard.js"></script>

스타일

{
    title: { 
        content: "차트의 타이틀", 
        font: "bold 32px 'Arial'", fontColor: "#999999" 
    },
    // 차트가 그려지는 부분(client)와 상/하/좌/우 영역의 크기 및 배경색을 지정한다.
    regions: {
        //  'h, w'는 pixel 단위 또는 %로 지정할 수 있다.
        header:	{ h: "0", bkcolor: "#fff" },
        left:	{ w: "0", bkcolor: "#fff" },
        right:	{ w: "0", bkcolor: "#fff" },
        footer:	{ h: "0", bkcolor: "#fff" },
        // 차트가 그려질 client 영역은 크기를 조절할 수 없다.
        client:	{ bkcolor: "#fff" }
    },
    // 범례 렌더링에 관한 스타일을 설정한다.
    category: {
        visible: false,
        // 위치: 'left'|'right'
        position: "left",
        font: "normal 11px 'Arial'", fontColor: "#000"
    },
    // X/Y 축 렌더링에 관한 스타일을 설정한다.
    axis: {
        x: {
            // X 축 이름
            name: "",
            font: "normal 11px 'Arial'", fontColor: "#000",
            marginLeft: 50, marginRight: 50,
            lineColor: "#aaa"
        },
        y: {
            // Y 축 이름
            name: "",
            font: "normal 11px 'Arial'", fontColor: "#000",
            marginTop: 50, marginBottom: 50,
            lineColor: "#aaa",
            // Y 축에서 값이 렌더링되는 간격
            step: 10,
            // Y 축 최상단 값
            maxValue: Number.MAX_VALUE,
            // 축 위치: 'left'|'right'
            position: "left"
        },
        grid: {
            lineColor: "#ddd",
            visible: true
        }
    },
    // 항목 렌더링에 관한 스타일을 설정한다.
    item: {
        // 그룹 간격, 0~1
        groupGapRatio: .8,
        // 항목 간격, 0~1 
        gapRatio: .8,
        // 라벨 출력 여부: true | false 
        labelVisible: true,
        // 라벨 회전: 'horizon'|'vertical'
        labelRotate: "horizon",
        // 값 출력 여부: false|true
        valueVisible: false,
        // 그룹 병합 여부
        groupMerging: false
    }
}

입력 데이터

입력받는 데이터는 다음과 같은 형식을 갖는 항목(JSON 객체)을 담는 배열이다.

{
    // (option) 항목을 묶는 그룹이다.
    group: "group_name",
    // (required) 항목의 이름이다.
    name: "item_name",
    // (required) 항목의 값이다.
    value: 100,
    // (option) 항목을 구분하는 색상으로서 #으로 시작하는 HEX 값이다.
    color: "#ff0000",
    // (option) 항목의 이름을 숨길
    nameVisible: true
}

데이터의 예는 다음과 같다.

 ];
