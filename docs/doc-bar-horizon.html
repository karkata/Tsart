가로-바-차트

개요

사용법

가로-바-차트를 사용하려면 tsart-bar-horizon.js 파일을 참조해야 한다.

<!-- Tsart 라이브러리 참조 -->
<script src="...path.../tsart.js"></script>
<script src="...path.../tsart-bar-horizon.js"></script>

스타일

{
    title: { 
        content: "차트의 타이틀", 
        font: "bold 32px 'Arial'", fontColor: "#999999" 
    },
    // 차트가 그려지는 부분(client)와 상/하/좌/우 영역의 크기 및 배경색을 지정한다.
    regions: {
        // 'h, w'는 pixel 단위 또는 %로 지정할 수 있다. 
        header:	{ h: "0", bkcolor: "#fff" },
        left:	{ w: "0", bkcolor: "#fff" },
        right:	{ w: "0", bkcolor: "#fff" },
        footer:	{ h: "0", bkcolor: "#fff" },
        // 차트가 그려질 client 영역은 크기를 조절할 수 없다.
        client:	{ bkcolor: "#fff" }
    },
    // 범례 렌더링에 관한 스타일을 설정한다.
    category: {
        visible: false,
        // 렌더링 대상: "item"|"group"
        target: "item",
        // 렌더링 위치: "left"|"right"
        position: "left",
        font: "normal 11px 'Arial'",
        fontColor: "#000"
    },
    axis: {
        x: {
            // X 축 이름 
            name: "",
            font: "normal 11px 'Arial'",
            fontColor: "#000",
            marginLeft: 50,
            marginRight: 50,
            lineColor: "#aaa",
            // X 축 간격
            step: 10,
            // X 축 최댓값 
            maxValue: Number.MAX_VALUE,
            // 축의 위치: "bottom"|"top"
            position: "bottom"
        },
        y: {
            // Y 축 이름
            name: "",
            font: "normal 11px 'Arial'",
            fontColor: "#000",
            marginTop: 50,
            marginBottom: 50,
            lineColor: "#aaa"
        },
        grid: {
            lineColor: "#ddd",
            visible: true
        }
    },
    item: {
        // 그룹 간격, 0 ~ 1
        groupGapRatio: .8,
        // 항목 간격, 0 ~ 1
        gapRatio: .8,
        // 항목 라벨 위치: true | false 
        labelVisible: true,
        // 항목 라벨 회전: 'horizon'|'vertical'
        labelRotate: "horizon",
        // 항목 값 출력: false | true
        valueVisible: false,
        // 값 출력 위치: out | in
        valuePosition: "out",
        // 그룹 병합
        groupMerging: false
    }
}

입력 데이터

입력받는 데이터는 다음과 같은 형식을 갖는 항목(JSON 객체)을 담는 배열이다.

{
    // (option) 항목을 묶는 그룹이다.
    group: "group_name",
    // (required) 항목의 이름이다.
    name: "item_name",
    // (required) 항목의 값이다.
    value: 100,
    // (option) 항목을 구분하는 색상으로서 #으로 시작하는 HEX 값이다.
    color: "#ff0000",
    // (option) 항목의 이름을 숨길
    nameVisible: true
}

데이터의 예는 다음과 같다.

const raw = [
    { group: "1반", name: "은영", value: 100, color: "#2eccfa" },
    { group: "4반", name: "한솔", value:  98, color: "#01df3a", nameVisible: false },
    // ... 데이터 생략
];

예제

가장 간단한 사용 예이다.

const raw = [
    { group: "1반", name: "은영", value: 100, color: "#2eccfa" },
    { group: "1반", name: "은서", value:  75, color: "#2eccfa" },
    { group: "2반", name: "윤주", value:  88, color: "#fe9a2e" },
    { group: "2반", name: "예은", value:  95, color: "#fe9a2e" },
    { group: "3반", name: "하원", value: 100, color: "#ff0080" },
    { group: "3반", name: "희진", value:  66, color: "#ff0080" },
    // 데이터 생략
 ];

Tsart.draw("#example-01", "bar-horizon", null, raw);

X 축 범위가 125까지 나타나는데, 이것은 내부 계산 방식 때문에 100 이상의 값까지 포함하기 때문이다. 그리고 가로 크기가 700px, 세로가 400px인 콘테이너 크기보다 차트의 크기가 좀 작아 보이는 것은 상/하/좌/우에 기본 마진이 존재하기 때문이다. 스타일을 부여해서 좀 더 다듬어 보자. 

Tsart.draw("#example-02", "bar-horizon", { 
    // 차트 타이틀을 넣기 위해서 20px 정도 높이를 주었다.
    regions: { header: { h: 20 } },
    title: { content: "학생 수학 성적", font: "bold 18px 'Open Sans'" },
    axis: { 
        y: { name: "(학생)" },
        // X 축 범위를 100까지 고정시켰고, 10단계로 정해서 10, 20, 30 ..., 100처럼 렌더링하도록 하였다.
        x: { name: "(점수)", maxValue: 100, step: 10 } 
    },
    item: { valueVisible: true } 
}, raw);

범례를 표현해보자.

Tsart.draw("#example-03", "bar-horizon", { 
    // 왼쪽 영역에 범례가 생성될 것이므로 left 영역이 보이도록 가로 크기를 100으로 설정했다.
    regions: { header: { h: 20 }, left: { w: 100 } },
    title: { content: "학생 수학 성적", font: "bold 18px 'Open Sans'" },
    // 범례가 보이도록 설정한다.
    category: { visible: true },
    axis: { y: { name: "(학생)" }, x: { name: "(점수)", maxValue: 100, step: 10 } },
    // 항목 라벨을 출력하지 않도록 하였다.
    item: { valueVisible: true, labelVisible: false } 
}, raw);

예제로 사용하는 표본 데이터가 각 항목별로 범례를 표현하기에는 어울리지 않는다. 그래서 항목 라벨 대신에 그룹 라벨을 범례로 사용하려고 한다.

Tsart.draw("#example-04", "bar-horizon", { 
    // 오른쪽 영역에 범례를 생성하는 것이 더 나아 보인다.
    regions: { header: { h: 20 }, left: { w: 100 } },
    title: { content: "학생 수학 성적", font: "bold 18px 'Open Sans'" },
    // 범례를 오른쪽에 생성하도록 하고, 대상을 그룹으로 바꿨다.
    category: { visible: true, target: "group", position: "right" },
    axis: { y: { name: "(학생)" }, x: { name: "(점수)", maxValue: 100, step: 10 } },
    item: { valueVisible: true } 
}, raw);

다음은 항목들을 그룹 단위로 합쳐서 차트에 표현하려고 한다. 그룹화를 하려면 항목마다 가지고 있는 group 속성에 구별할 그룹 이름을 지정해야 한다. 그리고 그룹에 속한 항목들을 하나로 합쳤을 때 제대로 표현이 되려면 그룹 항목들을 정렬해줘야 한다. tsart는 groupMerging 속성을 true로 설정하면 데이터를 등록할 때 내부적으로 그룹 내부 정렬을 수행한다. (그룹 머지를 하게 되면 막대의 값은 출력되지 않는다.)

// 각 그룹마다 성적 순으로 색상을 다르게 설정했다.
const raw2 = [
    { group: "1반", name: "은영", value: 100, color: "#2eccfa" },
    { group: "1반", name: "은서", value:  75, color: "#01df3a" },
    { group: "2반", name: "윤주", value:  88, color: "#fe9a2e" },
    { group: "2반", name: "예은", value:  95, color: "#2eccfa" },
    ...
];    
Tsart.draw("#example-05", "bar-horizon", { 
    ...
    // groupMerging 속성 값을 true로 설정해주었다.
    item: { groupMerging: true } 
}, raw2);
